15  set fmt [lindex [split $type _] 1]

17  } else {
18  if { $type == "float" || $type == "double"} {

21  set fmt 18.3f

24  }

26  string {
27  set record "[format %-8s $key]= '[format %-18s $value]'"

30  set record "[format %-8s $key]= [format %19d $value]"
