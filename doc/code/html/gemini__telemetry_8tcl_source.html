13 ## Documented proc \c geminiConnect .
14 # \param[in] scope Telescope id , north or south
15 #
16 # Connect to the telemery server
17 #
18 # Globals :
19 # GEMINICFG - Gemini specific proprties
20 #
21 proc geminiConnect { scope } {
22 global GEMINICFG
23  set GEMINICFG(site) $scope
24  set handle -1
25  set ok [ catch { set res [exec ping $GEMINICFG($scope,ip) -c 1] } ]
26  if { $ok == 0} {
27  catch {
28  set handle [socket $GEMINICFG($scope,ip) $GEMINICFG($scope,port)]
29  fconfigure $handle -buffering line -blocking 0
30  }
31  if { $handle < 0} {
32  set GEMINICFG(handle) -1
33  debuglog "Failed to connect to Gemini service at $GEMINICFG($scope,ip) port $GEMINICFG($scope,port) "
34  } else {
35  debuglog "Connected to Gemini $GEMINICFG($scope,ip) port $GEMINICFG($scope,port) - OK"
36  set GEMINICFG(handle) $handle
37  puts $GEMINICFG(handle) "get airmass"
38  gets $GEMINICFG(handle) rec
39  }
40  return $handle
41  } else {
42  set GEMINICFG(handle) -1
43  debuglog "Failed to connect to Gemini service at $GEMINICFG($scope,ip) port $GEMINICFG($scope,port) "
44  return -1
45  }
46 }

86  geminiConnect $GEMINICFG(site)
87  return
88  }
89  }
90  } else {
91  debuglog "Gemini telemetry lost"
92  simGeminiTelemetry
