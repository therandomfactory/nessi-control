24 ## Documented proc \c cAndorSetProperty .
25 # \param[in] arm Name of intrument arm red/blue
26 # \param[in] prop Camera property name
27 # \param[in] val Value to set the the property to
28 # \param[in] val2 Optional second value for HSSpeed usage
29 #
30 # This wraps the low level (C) shared library calls which set camera properties
31 # e.g it will call methods like SetExpsoureTime. It records the results of
32 # sucessfull calls in the ANDOR_CFG array for easy access from the tcl side
33 #
34 #
35 # Globals :\n
36 # ANDOR_CFG - Andor camera properties
37 # ANDOR_ARM - Instrument arm this camera is installed in red/blue
38 #
39 proc cAndorSetProperty { cam prop val {val2 ""} } {
40 global ANDOR_CFG ANDOR_ARM
41  set res "SetProperty failed"
42  catch {
43  if { $prop == "HSSpeed" } {
44  if { $val == 0 } { set res [andorSetProperty $cam HSSpeed 0 $val2] ; set prop EMHSSpeed}
45  if { $val == 1 } { set res [andorSetProperty $cam HSSpeed 1 $val2]}
46  set val $val2
47  } else {
48  set res [andorSetProperty $cam $prop $val]
49  }
50  if { $res == "" } {
51  set ANDOR_CFG($cam,$prop) $val
52  set ANDOR_CFG($ANDOR_ARM,$prop) $val
53  puts stdout "Setting ANDOR_CFG($ANDOR_ARM,$prop) to $val"
54  } else {
55  set ANDOR_CFG($cam,$prop) "fail -$res"
56  set ANDOR_CFG($ANDOR_ARM,$prop) "fail - $val"
57  puts stdout "ERROR Setting ANDOR_CFG($ANDOR_ARM,$prop) to $val"
58  }
59  }
60  return $res
61 }
