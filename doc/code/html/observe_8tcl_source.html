450 #
451 # This routine manages a sequence of exposures. It updates bias columns\n
452 # specifications in case they have been changed, then it loops thru\n
453 # a set of frames, updating the progress bar.
454 #
455 # Globals :
456 # SCOPE - Telescope parameters, gui setup
457 # CONFIG - Image geometry configuration
458 # LASTACQ - Type of last image acqusition
459 # ANDOR_DEF - Andor defaults
460 # ACQREGION - Sub-frame region coordinates
461 # ANDOR_CFG - Array of camera settings
462 # ANDOR_SOCKET - Andor camera server socket handles
463 # ANDOR_SHUTTER - Shutter mode names
464 # SCOPE - Telescope parameters, gui setup
465 # FWHEELS - Filter data
466 # OBSPARS - Default observation parameters
467 # STATUS - Exposure status
468 # DEBUG - Set to 1 for verbose logging
469 # TELEMETRY - Array of telemetry for headers and database usage
470 # DATAQUAL - Array of data quality information
471 # INSTRUMENT - Array of instrument configuration data
472 #
473 proc startsequence { {save keep} } {
474 global SCOPE OBSPARS FWHEELS STATUS DEBUG REMAINING LASTACQ TELEMETRY DATAQUAL SPECKLE_FILTER INSTRUMENT
475 global ANDOR_CCD ANDOR_EMCCD ANDOR_CFG ANDOR_SHUTTER
476 if { $ANDOR_CFG(videomode) == 0} {
477  set iseqnum 0
478  set STATUS(observing) 1
479  prepsequence
480  cameraStatuses
481  catch {
482  set x [andorSetControl 0 frame 0]
483  set x [andorSetControl 1 frame 0]
484  }
485  if { $SCOPE(exptype) == "Test"} {
486  exec rm -f $SCOPE(datadir)/TESTr.fits
487  exec rm -f $SCOPE(datadir)/TESTb.fits
488  commandAndor red "imagename TEST 1"
489  commandAndor blue "imagename TEST 1"
490  acquireTest
491  set STATUS(observing) 0
492  return
493  }
494  set autofilter [checkAutoFilter blue]
495  set rautofilter [checkAutoFilter red]
496  set FWHEELS(red,exposure) $SCOPE(exposure)
497  set FWHEELS(blue,exposure) $SCOPE(exposure)
498  set SCOPE(maxExposure) $SCOPE(exposure)
499  while { $autofilter != ""} {
500  set bfilter [lindex $autofilter 0]
501  if { $bfilter > 0} {
502  selectfilter blue $bfilter
503  commandAndor blue "filter $SPECKLE_FILTER(blue,current)"
504  commandAndor blue "emccdgain $FWHEELS(blue,$bfilter,emgain)"
505  set FWHEELS(blue,exposure) $FWHEELS(blue,$bfilter,exp)
506  set SCOPE(exposure) $FWHEELS(blue,$bfilter,exp)
507  commandAndor blue "setexposure $SCOPE(exposure)"
508  }
509  set rfilter [lindex $rautofilter 0]
510  if { $rfilter > 0} {
511  selectfilter red $rfilter
512  commandAndor red "filter $SPECKLE_FILTER(red,current)"
513  commandAndor red "emccdgain $FWHEELS(red,$rfilter,emgain)"
514  set FWHEELS(red,exposure) $FWHEELS(red,$rfilter,exp)
515  set SCOPE(exposureRed) $FWHEELS(red,$rfilter,exp)
516  commandAndor red "setexposure $SCOPE(exposureRed)"
517  if { $SCOPE(exposureRed) > $} {
518  set SCOPE(maxExposure) $SCOPE(exposureRed)
519  }
520  }
521  set autofilter [lrange $autofilter 1 end]
522  set rautofilter [lrange $rautofilter 1 end]
523  set iseqnum 0
524  updateRedisTelemetry mode observing
525  updateRedisTelemetry exposure $SCOPE(exptype)
526  set STATUS(abort) 0
527  while { $iseqnum < $SCOPE(numseq) && $STATUS(abort) == 0} {
528  .lowlevel.p configure -value 0
529  set ifrmnum 0
530  incr iseqnum 1
531  .lowlevel.seqp configure -value [expr $iseqnum*100/$SCOPE(numseq)]
532  .main.abort configure -bg orange -relief raised -fg black -command abortsequence
533  while { $ifrmnum < $SCOPE(numframes)} {
534  set clncmt [join [split [.main.comment get 0.0 end] "\`\"\'\[\]\{\}\&\%\$\\"] _]
