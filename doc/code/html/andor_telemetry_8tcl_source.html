16 # Update the items of telemetry related to camera specifics, used for populating\n

19 #
20 # Globals :\n
21 # ANDOR_CFG - Andor camera properties\n
22 # SPECKLE_DIR - Directory path to speckle code\n
23 # ANDOR_ARM - Instrument arm this camera is installed in red/blue
24 # ANDOR_ROI - Region of interest parameters\n
25 # CAM - Andor camera id used in the C code, 0 or 1\n
26 # TELEMETRY - Array of telemetry items for header/database usage
27 # SCOPE - Array of telescope information
28 # CAMSTATUS - Camera parameters
29 #
30 proc updateAndorTelemetry { arm } {
31 global ANDOR_CFG CAMSTATUS TELEMETRY SPECKLE_DIR SCOPE ANDOR_ROI CAM ANDOR_ARM
32  set state(0) "Off"
33  set state(1) "On"
34  set TELEMETRY(speckle.andor.head) "Andor iXon Emccd"
35  if { $ANDOR_CFG($CAM,OutputAmplifier) == 0} {
36  set TELEMETRY(speckle.andor.amplifier) "Electron Multiplying"
37  set TELEMETRY(speckle.andor.emccdmode) "On"
38  } else {
39  set TELEMETRY(speckle.andor.amplifier) "CCD"
40  set TELEMETRY(speckle.andor.emccdmode) "Off"
41  }
42  set TELEMETRY(speckle.andor.imagename) "[set ANDOR_CFG(imagename)][string range [set ANDOR_ARM] 0 0].fits"
43  set TELEMETRY(speckle.andor.acquisition_mode) "Single scan"
44  if { $TELEMETRY(speckle.andor.numberkinetics) > 1} { set TELEMETRY(speckle.andor.acquisition_mode) "Kinetics mode"}
45  set TELEMETRY(speckle.andor.int_time) $ANDOR_CFG($arm,ExposureTime)
46  set TELEMETRY(speckle.andor.kinetic_time) $ANDOR_CFG($CAM,TKinetics)
47  set TELEMETRY(speckle.andor.exptime) $ANDOR_CFG($CAM,TExposure)
48  set TELEMETRY(speckle.andor.numaccum) $ANDOR_CFG($arm,NumberAccumulations)
49  if { $TELEMETRY(speckle.andor.numaccum) > 1} { set TELEMETRY(speckle.andor.acquisition_mode) "Kinetics + Accumulate mode"}
50  set TELEMETRY(speckle.andor.accumulationcycle) $ANDOR_CFG($CAM,TAccumulate)
51  set TELEMETRY(speckle.andor.read_mode) "Frame transfer"
52  set TELEMETRY(speckle.andor.fullframe) "1,1024,1,1024"
53  set TELEMETRY(speckle.andor.frametransfer) $state($ANDOR_CFG($arm,FrameTransferMode))
54  set TELEMETRY(speckle.andor.biasclamp) $state($ANDOR_CFG($arm,BaselineClamp))
55  set TELEMETRY(speckle.andor.hbin) $ANDOR_CFG(binning)
56  set TELEMETRY(speckle.andor.roi) "[expr round($ANDOR_ROI(xs))] ,[expr round($ANDOR_ROI(xe))] ,[expr round($ANDOR_ROI(ys))], [expr round($ANDOR_ROI(ye))]"
57  set TELEMETRY(speckle.andor.vbin) $ANDOR_CFG(binning)
58  set TELEMETRY(speckle.andor.datatype) $ANDOR_CFG($arm,fitsbits)
59  set dll [string range [file tail [glob $SPECKLE_DIR/lib/libUSBI2C.so.*.0]] 13 end]
60  set TELEMETRY(speckle.andor.sw_version) $dll
61  set TELEMETRY(speckle.andor.exposure_total) [expr $ANDOR_CFG($arm,NumberKinetics) * $ANDOR_CFG($CAM,TExposure) * $ANDOR_CFG($arm,NumberAccumulations)]
62  set TELEMETRY(speckle.andor.em_gain) $ANDOR_CFG($arm,EMCCDGain)
63  set TELEMETRY(speckle.andor.preamp_gain) [expr $ANDOR_CFG($arm,PreAmpGain) +1]
64  set TELEMETRY(speckle.andor.serial_number) $ANDOR_CFG($CAM,SerialNumber)
65  set TELEMETRY(speckle.andor.target_temperature) $ANDOR_CFG($arm,Temperature)
66  set TELEMETRY(speckle.andor.ccdtemp) $ANDOR_CFG(ccdtemp)
67  set TELEMETRY(speckle.andor.prescans) 0
68  set TELEMETRY(speckle.andor.vertical_speed) $ANDOR_CFG(VSSpeed,$ANDOR_CFG($CAM,VSSpeed))
69  if { $ANDOR_CFG($CAM,OutputAmplifier) == 0} {
70  set TELEMETRY(speckle.andor.horizontal_speed) [lindex $ANDOR_CFG(emccdhorizontalspeeds) $ANDOR_CFG($CAM,EMHSSpeed)]
71  } else {
72  set TELEMETRY(speckle.andor.horizontal_speed) [lindex $ANDOR_CFG(ccdhorizontalspeeds) $ANDOR_CFG($CAM,HSSpeed)]
73  }
74  set TELEMETRY(speckle.andor.bias_estimate) $ANDOR_CFG(bias)
75  set TELEMETRY(speckle.andor.peak_estimate) $ANDOR_CFG(peak)
76  set now [exec date -u +%Y-%m-%dT%H:%M:%S.0]
77  set TELEMETRY(speckle.scope.recid) [set SCOPE(obsdate)]_$SCOPE(telescope)"
78  set TELEMETRY(speckle.andor.read_mode) "Image"
79  set TELEMETRY(speckle.scope.site) $SCOPE(telescope)
80  set TELEMETRY(speckle.scope.ProgID) $SCOPE(ProgID)
81 }
