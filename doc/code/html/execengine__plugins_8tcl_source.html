9 proc observingEngine { op {sname ""} } {
10 global SPECKLE_EEDIR SPECKLE_EECMDS SPECKLE_EEFILE SPECKLE_EEINDEX SPECKLE_EE
11  set result 0
12  switch $op {
13  load {
14  if { $sname == "" } {
15  set fsel [tk_getOpenFile -initialdir $SPECKLE_EEDIR -filetypes {{{Observing-scripts} {.obs}}}]
16  } else {
17  set fsel $sname
18  }
19  if { [file exists $fsel] } {
20  set SPECKLE_EEFILE $fsel
21  set SPECKLE_EE idle
22  set SPECKLE_EECMDS [split [exec cat $SPECKLE_EEFILE] \n]
23  set SPECKLE_EEINDEX 0
24  set result [obsEngValidate $SPECKLE_EECMDS]
25  if { $result } {
26  obsEngControls show
27  }
28  }
29  }
30  cancel {
31  obsEngControls hide
32  set SPECKLE_EE halt
33  set SPECKLE_EECMDS ""
34  set SPECKLE_EEFILE ""
35  }
36  pause {
37  set SPECKLE_EE pause
38  obsEngControls pause
39  }
40  resume -
41  run {
42  set SPECKLE_EE running
43  obsEngControls resume
44  obsEngRun
45  }
46  step {
47  set SPECKLE_EE step
48  obsEngControls resume
49  obsEngRun
50  }
51  }
52  return $result
53 }

104 proc obsEngNext { {lookahead 0} } {
105 global SPECKLE_EEINDEX SPECKLE_EECMDS SPECKLE_EEDURATION
106  if { $SPECKLE_EEINDEX > [llength $SPECKLE_EECMDS]} {return -1}
107  set gotnext 0
108  while { $gotnext == 0} {
109  set nxtcmd [lindex $SPECKLE_EECMDS $SPECKLE_EEINDEX]
110  if { [lindex $nxtcmd 0] == "durationEstimate" && $lookahead == 0} {
111  set nothanks [obsEngDuration [lindex $nxtcmd 1]]
112  if { $nothanks} {obsEngControls show; return -1}
113  } else {
114  if { [string trim $nxtcmd] != ""} {
115  if { [string range $nxtcmd 0 0] != "#"} {
116  set gotnext 1
117  incr SPECKLE_EEINDEX 1
118  }
119  }
120  }
121  if { $gotnext == 0} {incr SPECKLE_EEINDEX 1}
122  if { $SPECKLE_EEINDEX > [llength $SPECKLE_EECMDS]} {
123  return -1
124  }
125  }
126  return $nxtcmd
127 }

183 proc obsEngValidate { obscmds } {
184 global CLI_SCRIPTCMDS DEBUG
185  set valid 1
186  set reason ""
187  foreach cmd $obscmds {
188  if { [string trim $cmd] != ""} {
189  if { [string range $cmd 0 0] != "#"} {
190  if {$DEBUG} {puts stdout "obsEngValidate Syntax check $cmd"}
191  set chk [string trim $cmd]
192  if { [lsearch $CLI_SCRIPTCMDS [lindex $chk 0]] < 0} {
193  set valid 0
194  set reason "${reason}Invalid command - [lindex $chk 0]\n"
195  }
196  }
197  }
198  }
199  if { $valid == 0} {
200  set it [tk_dialog .d "Syntax error" "Bad .obs file\n$reason" {} -1 "OK"]
201  }
202  return $valid
203 }

224 proc waitforFocus { target timeleft } {
225  while { $timeleft > 0} {
226  set posa [lindex [wiyn info oss.secondary.posa] 0]
227  set posb [lindex [wiyn info oss.secondary.posb] 0]
228  set posc [lindex [wiyn info oss.secondary.posc] 0]
229  set current -99990.
230  catch {set current [expr ($posa+$posc)/2.0/8.0 ]}
231  if { [expr abs($target-$current)] < 10} {
232  return 0
233  }
234  after 1000
235  incr timeleft -1
236  update
237  }
