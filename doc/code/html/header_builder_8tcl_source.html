20 # \param[in] from File to read stream definitions from
21 #
22 #
23 # Globals :\n
24 # TELEMETRY - Array of telemetry items for headers and database usage\n
25 # STREAM - Array of telemetry stream names\n
26 # PDEBUG - Debug verbosity\n
27 # FITSKEY - Array of FITS keywords\n
28 # FITSTXT - Array of FITS header descriptions
29 #
30 proc loadstreamdefs { {from telem.conf} } {
31 global TELEMETRY STREAMS PDEBUG FITSKEY FITSTXT
32  set fin [open $from r]
33  while { [gets $fin rec] > -1} {
34  if { [string trim $rec] != ""} {
35  if { [string range $rec 0 0] != "#"} {
36  if { [lindex $rec 0] == "stream" || [lindex $rec 0] == "category"} {
37  lappend STREAMS [lindex $rec 1]
38  set cstream [lindex $rec 1]
39  if { $PDEBUG} {debuglog "Definition of $cstream"}
40  } else {
41  set TELEMETRY($cstream.[lindex $rec 0]) "NA"
42  set FITSKEY($cstream.[lindex $rec 0]) [lindex $rec 1]
43  set FITSTXT($cstream.[lindex $rec 0]) [lrange $rec 2 end]
44  if { $PDEBUG} {debuglog " parameter $cstream.[lindex $rec 0]"}
45  }
46  }
47  }
48  }
49  close $fin
50 }

86 ## Documented proc \c fillheader .
87 # \param[in] args Optional arguments , scope-instrument identifier
88 #
89 # Create a FITS header structure with all the required keywords
90 #
91 # Globals :\n
92 # TELEMETRY - Array of telemetry items for headers and database usage\n
93 # SEQNUM - Sequence number\n
94 # CACHETELEMETRY - Cache of telemetry items\n
95 # PDEBUG - Debug verbosity\n
96 # HEADERS - Names of header types\n
97 # FITSKEY - Array of FITS keywords\n
98 # FITSTXT - Array of FITS header descriptions\n
99 # FROMSTARTEXP - Array of values to use from exposure start
100 # ANDOR_ARM - Instrument arm, red or blue
101 #
102 proc fillheader { args } {
103 global TELEMETRY PDEBUG HEADERS FITSKEY FITSTXT SEQNUM SCOPE
104 global FROMSTARTEXP CACHETELEMETRY ANDOR_ARM env
105  set fhead ""
106  set gottelem "no"
107  catch {
108  speckleTelemetryUpdate
109  set gottelem yes
110  }
111  if { $gottelem == "no"} {
112  debuglog "****************** NO TELEMETRY AVAILABLE ***********************"
113  debuglog "****************** PLEASE CHECK THE VII IS RUNNING *************"
114  debuglog "****************** AND MAX CONNECTIONS IS >5 ********************"
115  simGeminiTelemetry
116  }
117  updateAndorTelemetry $ANDOR_ARM
118  set type unknown
119  if { $env(TELESCOPE) == "GEMINI"} {set type gemini-speckle}
120  if { $env(TELESCOPE) == "WIYN"} {set type wiyn-speckle}
121  showTelemetry
122  set fhead "[fitshdrrecord HDR_REV string {4.00 10-Feb-2020q} Header-Rev ]\n"
123  foreach i $HEADERS($type) {
124  if { $PDEBUG > 1} {debuglog "processing $i"}
125  if { [info exists FITSKEY($i)]} {
126  set key $FITSKEY($i)
127  set text $FITSTXT($i)
128  } else {
129  set key [string toupper [lindex [split $i .] 2] ]
130  set text "from $i"
131  }
132  if { $key == "PROC"} {
133  set cmd "[lindex $text 0] \"$TELEMETRY($i)\""
134  set key [lindex $text 1]
135  set text [lrange $text 2 end]
136  set type string
137  catch {set type "$TELEMETRY($i,t)"}
138  set value "$TELEMETRY($i)"
139  if { [info exists FROMSTARTEXP($key)]} {
140  set value $CACHETELEMETRY($i)
141  }
142  if { $value == "Attribute" && $key != "FOCUS"} {
143  set type string
144  set value "Not available"
145  } else {
146  set value NA
147  catch {set parse [eval $cmd]}
148  set value [lrange $parse 1 end]
149  set type [lindex $parse 0]
150  }
151  } else {
152 ### set current [$TOMPG info $i]
153 ### for testing
154  set value NA
155  catch {set value "$TELEMETRY($i)"}
156 # if { [info exists FROMSTARTEXP($key)] } {
157 # set value $CACHETELEMETRY($i)
158 # }
159  set type string
160  if { $SCOPE(telescope) == "wiyn"} {
161  catch {set type "$TELEMETRY($i,t)"}
162  if { $value == "Attribute"} {
163  set type string
164  set value "Not available"
165  }
166  }
167  }
168  set new [fitshdrrecord $key $type $value $text ]
169  if { [string trim [lindex [split $new "/"] 0]] != ""} {
170  if { $PDEBUG > 1} {debuglog "HEADER->$new"}
171  set fhead "$fhead$new\n"
172  } else {
173  debuglog "No value for $key"
174  }
175  }
176  set fhead "$fhead[fitshdrrecord OBSID string [obsid] Observation-ID]\n"
177  set fhead "$fhead[fitshdrrecord GEMPRGID string [obsid] Observation-ID]\n"
178  set fhead "$fhead[fitshdrrecord SPKLESEQ integer $SEQNUM SPECKLENFO-sequence]\n"
179  set fhead "$fhead[fitshdrrecord SPKLEDAT string [lrange [exec date] 1 3] SPECKLENFO-timestamp]"
180  if { [string trim [lindex $args 1]] != ""} {
181  set fdbg [open [lindex $args 1] w]
182  puts $fdbg "$fhead"
183  close $fdbg
184  set fhead "OK"
185  }
186  incr SEQNUM 1
187  return "$fhead"
188 }

268 ## Documented proc \c fitshdrrecord .
269 # \param[in] key A FITS header keyword
270 # \param[in] type Type of the information, integer, float, double, string
271 # \param[in] value The data value
272 # \param[in] text The text description of the item
273 #
274 # Format a FITS header record
275 #
276 proc fitshdrrecord { key type value text } {
277  set record ""
278  set v1 [lindex $value 0]
279  set fmt 18.4f
280  if { [llength [split $type _]] > 1} {
281  set fmt [lindex [split $type _] 1]
282  set type [lindex [split $type _] 0]
283  }
284  if { $value == "NA"} {set type string}
285  switch $type {
286  string {
287  set record "[format %-8s $key]= [format %18s '[string trim $value]']"
288  }
289  integer {
290  set record "[format %-8s $key]= [format %19d [expr int($value)]]"
291  }
292  boolean {
293  set record "[format %-8s $key]= [format %19s $value]"
294  }
295  float {
296  if { [expr abs($value)] < 10000 } {set fmt 18.6f}
297  set record [string toupper "[format %-8s $key]= [format %$fmt $v1]"]
298  }
299  bigfloat {
300  if { [expr abs($value)] < 10000 } {set fmt 18.9f}
301  set record [string toupper "[format %-8s $key]= [format %$fmt $v1]"]
302  }
303  double {
304  if { [expr abs($value)] < 10000 } {set fmt 18.6f}
305  if { [expr abs($value)] < 1 } {set fmt 18.12f}
306  set record [string toupper "[format %-8s $key]= [format %$fmt $v1]"]
307  }
308  }
309  set record "$record /[format %-48s $text]"
310 # if { $TOMPG == "wiyn" } {
311 # set record "|TELESCOPE seqstar $record"
312 # }
313  return "$record"
314 }

317 ## Documented proc \c appendHeader .
318 # \param[in] imgname Name of FITS file
319 #
320 # Add the FITS header information to a file
321 #
322 # Globals :\n
323 # TELEMETRY - Array of telemetry items for headers and database usage\n
324 # SCOPE - Array of telescope parameters\n
325 # env - Environment variables\n
326 # SPECKLEHDRLOG - Cache of header details
327 #
328 proc appendHeader { imgname } {
329 global SPECKLEHDRLOG SCOPE env TELEMETRY INSTRUMENT
330  set hdr [fillheader $env(TELESCOPE)-$SCOPE(instrument)]
331  puts $SPECKLEHDRLOG "$hdr"
332  set fid [fits open $imgname]
333  foreach rec [split $hdr \n] {
334  $fid put keyword "$rec"
335  }
336  if { [info proc headerStandard] == "headerStandard"} {
337  headerStandard $fid
338  }
339  if { [info proc headerGeometry] == "headerGeometry"} {
340  headerGeometry $fid
341  }
342  if { [info proc headerAstrometry] == "headerAstrometry"} {
343  headerAstrometry $fid $TELEMETRY(tcs.telescope.ra) $TELEMETRY(tcs.telescope.dec)
344  }
345  if { [info proc headerComments] == "headerComments"} {
346  fits close $fid
347  set fid [fits open $imgname]
348  headerComments $fid
349  }
350  fits close $fid
351 }

364 ## Documented proc \c calculateXSEC .
365 # \param[in] name Geometry section name
366 #
367 # Calculate image geometry header data
368 #
369 #
370 # Globals :\n
371 # SCOPE - Array of telescope parameters\n
372 # IMGMETA - Image geometry metadata
373 #
374 proc calculateXSEC { name } {
375 global IMGMETA SCOPE
376  set imgcols [expr int($IMGMETA(imageCols,postvalue))]
377  set imgrows [expr int($IMGMETA(imageRows,postvalue))]
378  set scicols [expr int($IMGMETA(sciCols,postvalue))]
379  set scirows [expr int($IMGMETA(sciRows,postvalue))]
380  set oricols [expr int($IMGMETA(pxlCols,postvalue)/$SCOPE(colBin))]
381  set orirows [expr int($IMGMETA(pxlRows,postvalue)/$SCOPE(rowBin))]
382  switch $name {
383  DATASEC {
384  set sec "\[1:$scicols,1:$scirows\]"
385  }
386  ORIGSEC {
387  set sec "\[1:$oricols,1:$orirows\]"
388  }
389  CCDSEC {
390 #must change to scistart:sciend for subrasters
391 # maybe xStart:scicols,yStart:scirows
392  set sec "\[1:$scicols,1:$scirows\]"
393  }
394  TRIMSEC {
395  set sec "\[1:$scicols,1:$scirows\]"
396  }
397  BIASSEC {
398  set startx [expr int($IMGMETA(sciCols,postvalue)+$IMGMETA(xPreScan,postvalue)+3)]
399  set endx [expr int($IMGMETA(imageCols,postvalue)-3)]
400  set sec "\[$startx:$endx,1:$scirows\]"
401  }
402  }
403  return "$sec"
404 }
