19 # POWER - Array of power strip configuration parameters

21 # SIMPOWER - 1 if we are simulation power strip control

23 proc powerPortSwitch { port state } {
24 global POWER SIMPOWER env
25  set user [string tolower $env(INSTRUMENT)]
26  set pw speckle
27  set ip $POWER($user,ip)
28  debuglog "Power cycling plug $port : new state = $state"
29  if { $SIMPOWER == 0} {
30  exec wget http://$ip/api/v2/config/powerplug --user=$user --password=$pw --post-data='\{"plug":$port, "state": "$state"\}'
31  }
32 }
