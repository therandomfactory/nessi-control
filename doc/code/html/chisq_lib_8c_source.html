257  * pseud2d -
258  *
259  *---------------------------------------------------------------------*/
260 static double pseud2d (int ix, int iy, double *a, double *fa)
261 {
262  double x, y, tx, ty, t;
263  double dddt, pexp, denom;
264 
265  x = ix - a[1];
266  y = iy - a[2];
267  tx = (1.0 / a[4]) * x;
268  ty = (1.0 / a[4]) * y;
269  t = HALF * (tx * x + ty * y);
270 
271  if (t > 0.0) {
272  denom = 1 + t + (t * t * HALF) + (THIRD * HALF * t * t * t);
273  dddt = 1 + t + (t * t * HALF);
274  pexp = 1 / denom;
275  }
276  else {
277  if (t < EXPMIN)
278  t = EXPMIN;
279  pexp = exp (-t);
280  denom = 1.0;
281  dddt = 1.0;
282  }
283 
284  fa[3] = pexp;
285  pexp = a[3] * pexp;
286  fa[4] = pexp * dddt / denom;
287  fa[1] = tx * fa[4];
288  fa[2] = ty * fa[4];
289  fa[4] = HALF * (tx * tx + ty * ty) * fa[4];
290  fa[0] = 1.0;
291 
292  return (pexp + a[0]);
293 }
