208 proc setOrielFilter { id n } {
209  oriel_write_cmd $id FILT?
210  after 200
211  set res [oriel_read_result $id]
212  set cpos [string range $res 4 4]
213  set npos [expr $n - $cpos]
214  while { $npos > 0} {
215  oriel_write_cmd $id NEXT
216  incr npos -1
217  after 2000
218  set res [oriel_read_result $id]
219  }
220  while { $npos < 0} {
221  oriel_write_cmd $id PREV
222  incr npos 1
223  after 2000
224  set res [oriel_read_result $id]
225  }
226  oriel_write_cmd $id FILT?
227  after 200
228  set res [oriel_read_result $id]
229  set cpos [string range $res 4 4]
230  if { $cpos == $n} {
231  return $n
232  }
233  return -1
234 }
