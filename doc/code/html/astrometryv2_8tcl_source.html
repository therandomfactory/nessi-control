14 # \param[in] ra RA coordinate
15 # \param[in] dec DEC coordinate
16 #
17 # Update the DS9 World Coordinate system based upon current RA,DEC
18 #
19 #
20 # Globals :\n
21 # ANDOR_CFG - Andor camera properties\n
22 # ANDOR_ARM - Instrument arm this camera is installed in red/blue
23 # ACQREGION - Region of interest parameters\n
24 # SCOPE - Array of telescope information
25 # CAMSTATUS - Camera parameters
26 # PI - pi
27 # DS9 - Name of a ds9 executable , ds9red or ds9blue
28 #
29 proc updateds9wcs { ra dec } {
30 global SCOPE ACQREGION PSCALES ANDOR_CFG PI DS9 ANDOR_ARM env TELEMETRY WCSPARS
31  readWCSpars $ANDOR_ARM $TELEMETRY(speckle.andor.inputzaber)
32  set radeg [expr ([hms_to_radians $ra]+[hms_to_radians 00:00:$WCSPARS(DELTARA)/15.])*180/$PI]
33  set decdeg [expr ([dms_to_radians $dec]+[dms_to_radians 00:00:$WCSPARS(DELTADEC)])*180/$PI]
34  set fout [open /tmp/[set ANDOR_ARM]wcs.wcs w]
35  if { $WCSPARS(CRVAL1) == "RA"} {
36  puts $fout "CRVAL1 $radeg"
37  puts $fout "CRVAL2 $decdeg"
38  puts $fout "CTYPE1 'RA--TAN'"
39  puts $fout "CTYPE2 'DEC--TAN'"
40  } else {
41  puts $fout "CRVAL1 $decdeg"
42  puts $fout "CRVAL2 $radeg"
43  puts $fout "CTYPE1 'DEC--TAN'"
44  puts $fout "CTYPE2 'RA--TAN'"
45  }
46  puts $fout "CDELT1 $WCSPARS(CDELT1)"
47  puts $fout "CDELT2 $WCSPARS(CDELT2)"
48  puts $fout "CROTA2 $WCSPARS(CROTA2)"
49  puts $fout "CRPIX1 [expr $ACQREGION(geom)/$ANDOR_CFG(binning)]"
50  puts $fout "CRPIX2 [expr $ACQREGION(geom)/$ANDOR_CFG(binning)]"
51  puts $fout "CD1_1 [expr $WCSPARS(CDELT1) * [cosd $WCSPARS(CROTA2)]]"
52  puts $fout "CD1_2 [expr -1.0*$WCSPARS(CDELT2) * [sind $WCSPARS(CROTA2)]]"
53  puts $fout "CD2_1 [expr $WCSPARS(CDELT1) * [sind $WCSPARS(CROTA2)]]"
54  puts $fout "CD2_2 [expr $WCSPARS(CDELT2) * [cosd $WCSPARS(CROTA2)]]"
55  puts $fout "WCSNAME 'FK5'"
56  puts $fout "RADECSYS 'FK5'"
57  puts $fout "EQUINOX 2000."
58  close $fout
59  exec xpaset -p $DS9 wcs replace /tmp/[set ANDOR_ARM]wcs.wcs
60 }

128 ## Documented proc \c dms_to_radians .
129 # \param[in] dms dd:mm:ss for conversion
130 #
131 # Convert ddd:mm:ss to radians
132 #
133 #
134 # Globals :
135 # PI - pi
136 #
137 proc dms_to_radians { dms } {
138 global PI
139  set t [string trim $dms "+ "]
140  set s 1
141  if { [string range $t 0 0] == "-"} {
142  set s -1
143  set dms [string trim $dms "-"]
144  }
145  set f [split $dms ":"]
146  set r [expr $s * ([scan [lindex $f 0] %d] + [scan [lindex $f 1] %d]/60.0 + [scan [lindex $f 2] %d]/3600.0 )/180. * $PI]
147 }

174 ## Documented proc \c readWCSpars .
175 #
176 #
177 # Convert hh:mm:ss to radians

179 #

181 # WCSPARS - User tunable WCS parameters from file
182 #
183 proc readWCSpars { arm mode } {
184 global env WCSPARS
185  if { $mode != "wide" && $mode != "speckle"} {
186  debuglog "WARNING: input zaber position unknown, using previous WCSPARS"
187  } else {
188  switch $env(TELESCOPE)_$env(GEMINISITE)_$mode {
189  GEMINI_north_speckle { set wcspars $env(SPECKLE_DIR)/wcsPars.$arm.speckle.geminiN }
190  GEMINI_south_speckle { set wcspars $env(SPECKLE_DIR)/wcsPars.$arm.speckle.geminiS }
191  WIYN_NA_speckle { set wcspars $env(SPECKLE_DIR)/wcsPars.$arm.speckle.wiyn }
192  GEMINI_north_wide { set wcspars $env(SPECKLE_DIR)/wcsPars.$arm.wide.geminiN }
193  GEMINI_south_wide { set wcspars $env(SPECKLE_DIR)/wcsPars.$arm.wide.geminiS }
194  WIYN_NA_fullframe { set wcspars $env(SPECKLE_DIR)/wcsPars.$arm.wide.wiyn }
195  }
196  set fin [open $wcspars r]
197  while { [gets $fin rec] > -1} {
198  set par [string trim [lindex [split $rec =] 0]]
199  set val [string trim [lindex [split $rec =] 1]]
200  set WCSPARS($par) $val
201  }
202  close $fin
203  }
204 }
