122 # Draw3DFunction --
123 # Plot a function of x and y
124 # Arguments:
125 # w Name of the canvas
126 # function Name of a procedure implementing the function
127 # Result:
128 # None
129 # Side effect:
130 # The plot of the function - given the grid
131 #
132 proc ::Plotchart::Draw3DFunction { w function } {
133  variable scaling
134 
135  set nxcells $scaling($w,nxcells)
136  set nycells $scaling($w,nycells)
137  set xmin $scaling($w,xmin)
138  set xmax $scaling($w,xmax)
139  set ymin $scaling($w,ymin)
140  set ymax $scaling($w,ymax)
141  set dx [expr {($xmax-$xmin)/double($nxcells)}]
142  set dy [expr {($ymax-$ymin)/double($nycells)}]
143 
144  foreach {fill border} $scaling($w,colours) {break}
145 
146  #
147  # Draw the quadrangles making up the plot in the right order:
148  # first y from minimum to maximum
149  # then x from maximum to minimum
150  #
151  for { set j 0} { $j < $nycells} { incr j} {
152  set y1 [expr {$ymin + $dy*$j}]
153  set y2 [expr {$y1 + $dy}]
154  for { set i $nxcells} { $i > 0} { incr i -1} {
155  set x2 [expr {$xmin + $dx*$i}]
156  set x1 [expr {$x2 - $dx}]
157 
158  set z11 [$function $x1 $y1]
159  set z12 [$function $x1 $y2]
160  set z21 [$function $x2 $y1]
161  set z22 [$function $x2 $y2]
162 
163  foreach {px11 py11} [coords3DToPixel $w $x1 $y1 $z11] {break}
164  foreach {px12 py12} [coords3DToPixel $w $x1 $y2 $z12] {break}
165  foreach {px21 py21} [coords3DToPixel $w $x2 $y1 $z21] {break}
166  foreach {px22 py22} [coords3DToPixel $w $x2 $y2 $z22] {break}
167 
168  $w create polygon $px11 $py11 $px21 $py21 $px22 $py22 \
169  $px12 $py12 $px11 $py11 \
170  -fill $fill -outline $border
171  }
172  }
173 }
