629

635  struct rotPoint *point1 = NULL;
636  struct rotPoint *point2 = NULL;
637  double dxCalc, dxMeas, thetaCalc;
638  double dyCalc, dyMeas, thetaMeas;
639 
640  currentPair = rot->nextPair;
641 
642  while ( (currentPair = currentPair->nextPair) != NULL ){
643 
644  point1 = currentPair->point1;
645  point2 = currentPair->point2;
646 
647  /*
648  ** check that the two points exists
649  ** and that they should be allowed
650  ** to enter into the calculation
651  ** flag pairs that are not valid as
652  ** deleted
653  */
654  if ( (point1 != NULL) && (point2 != NULL) &&
655  (currentPair->id > 0) ){
656 
657  /*
658  ** find the two legs of each
659  ** triangle-measured and calculated
660  */
661  dxCalc = point1->starLoc.x - point2->starLoc.x;
662  dyCalc = point1->starLoc.y - point2->starLoc.y;
663 
664  dxMeas = point1->measLoc.x - point2->measLoc.x;
665  dyMeas = point1->measLoc.y - point2->measLoc.y;
666 
667  currentPair->rSep2 = ((dxCalc*dxCalc) + (dyCalc*dyCalc) ) / 1.0e9;
668 
669  /*
670  ** find the absolute angles
671  ** of each line wrt x axis
672  */
673  thetaCalc = normalize(atan2(dyCalc, dxCalc));
674  thetaMeas = normalize(atan2(dyMeas, dxMeas));
675 
676  /*
677  ** now find the rotation angle
678  ** for this pair
679  */
680  currentPair->rotAngle = normalize(thetaMeas - thetaCalc);
681  }
682  }
683
