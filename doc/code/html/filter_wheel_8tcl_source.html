238 ## Documented proc \c setOrielFilter .
239 # \param[in] id Wheel identifier
240 # \param[in] n Filter position
241 #
242 # Call low level code to move filter to selected position
243 #
244 proc setOrielFilter { id n } {
245  set res -1
246  catch {
247  oriel_write_cmd $id FILT?
248  after 500
249  set res [oriel_read_result $id]
250  set cpos [string range $res 4 4]
251  set npos [expr $n - $cpos]
252  while { $npos > 0 } {
253  oriel_write_cmd $id NEXT
254  incr npos -1
255  after 2000
256  set res [oriel_read_result $id]
257  }
258  while { $npos < 0 } {
259  oriel_write_cmd $id PREV
260  incr npos 1
261  after 2000
262  set res [oriel_read_result $id]
263  }
264  oriel_write_cmd $id FILT?
265  after 500
266  set res [oriel_read_result $id]
267  }
268  set cpos [string range $res 4 4]
269  if { $cpos == $n} {
270  return $n
271  }
272  return -1
273 }
